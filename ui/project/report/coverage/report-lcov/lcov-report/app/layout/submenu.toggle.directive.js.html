    angular
        .module('app.layout')

    function toggleSubmenu($window) {
        // Usage:

        var directive = {
            link: link,
            restrict: 'A'
        };
        return directive;
 
        function link(scope, element, attrs) {
            element.click(function () {
                element.parent().toggleClass('toggled');
                element.parent().find('ul').stop(true, false).slideToggle(200);
            });
        }
    }
 
})();
